```
2^19 = 2^9 * 2^9 * 2 = 512 * 512 * 2 = 262144

19 = 3 + 4 * 4 = 0x80000

16384 = 512 * 32 = 2^9 * 2^5 = 2^14

14 = 2 + 4 * 3 = 0x4000

0x10000 = 0 + 4 * 4 = 16

2^16 = 2^10 * 2^6 = 1024 * 64 = 65536

2^17 = 65536 * 2 = 131072

17 = 1 + 4 * 4 = 0x20000

32 = 2^5

5 = 1 + 4 * 1 = 0x20

0x80 = 3 + 4 * 1 = 7

2^7 = 128
```
